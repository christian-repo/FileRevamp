    Create src/FileRevamp/Output/Reporter.cs:
      Namespace: FileRevamp.Output
      Class Reporter
        Constructor: Reporter() — no dependencies (formats strings, caller writes to console)
        
        Public method: string FormatResultLine(RenameResult result)
          switch result.Status:
            DryRun: return $"[DRY RUN] {result.OriginalName} → {result.NewName}"
            Renamed: return $"{result.OriginalName} → {result.NewName}"
            Skipped: return $"SKIP {result.OriginalName}: {result.FailureReason}"
            Failed: return $"FAIL {result.OriginalName}: {result.FailureReason}"
          
        Public method: string FormatSummary(IEnumerable<RenameResult> results)
          var list = results.ToList()
          int renamed = list.Count(r => r.Status == RenameStatus.Renamed)
          int failed = list.Count(r => r.Status == RenameStatus.Failed)
          int skipped = list.Count(r => r.Status == RenameStatus.Skipped)
          return $"Renamed: {renamed}  Failed: {failed}  Skipped: {skipped}"
        
        Public method: string FormatDryRunComplete()
          return "Dry run complete — 0 files modified."

        Public static method: string? ValidateOutputName(string filename)
          if string.IsNullOrEmpty(filename): return $"Computed filename is empty"
          if filename.EndsWith('.'): return $"Computed filename has trailing dot: '{filename}'"
          if filename.EndsWith(' '): return $"Computed filename has trailing space: '{filename}'"
          return null

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| RenameResult → Reporter → AnsiConsole | Filenames containing Spectre markup chars ([ ]) could inject markup if not escaped |
| CLI args → CommandAppTester | Test harness receives same args as production; validates security controls work end-to-end |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | Reporter → AnsiConsole output — filenames with [ ] chars could inject Spectre markup | mitigate | Wrap all Reporter.FormatResultLine output in Markup.Escape() before passing to AnsiConsole.MarkupLine — implemented in RenameCommand.Execute |
| T-03-02 | Information Disclosure | Summary line reveals failure count to console | accept | Failures are about the user's own files; this disclosure is intentional and required by RPRT-02 |
| T-03-03 | Denial of Service | Very large directory with 100,000+ files materializes all RenameResults into a List | mitigate | RenameCommand uses orchestrator.Execute().ToList() which buffers all results. For Phase 1 (flat directory, non-recursive), this is bounded and acceptable. Phase 2 will address streaming. Note in code: "TODO(Phase 2): stream results for large batches" |
| T-03-SC | Tampering | Spectre.Console.Cli.Testing 0.55.0 NuGet package | mitigate | Package is from the official Spectre.Console project (spectreconsole.net), confirmed [VERIFIED] per STACK.md research. Same vendor and version as the production Spectre.Console.Cli dependency. |
</threat_model>
